// index.html / kibou-kyuu.html のLEAVE_TYPESとコードを揃えてある const LEAVE_TYPE_CODES = ['off','paid','paid_am','paid_pm','flex','child','child_am','child_pm','bereave','absence','absence_am','absence_pm']; function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'saveStaff') { let s = ss.getSheetByName('スタッフ') || ss.insertSheet('スタッフ'); s.clearContents(); s.appendRow(['id','name','gender','type','days']); data.staff.forEach(r => s.appendRow([r.id, r.name, r.gender, r.type, r.days.join(',')])); } if (action === 'saveShifts') { // 保存のたびにシート全体を消すと他の月のデータが消えるため、 // 対象の年月の行だけを入れ替え、他の月は残す。 let s = ss.getSheetByName('シフト') || ss.insertSheet('シフト'); if (s.getLastRow() === 0) { s.appendRow(['year','month','staffId','day','value']); } if (data.shifts.length > 0) { const targetY = Number(data.shifts[0].year); const targetM = Number(data.shifts[0].month); const values = s.getDataRange().getValues(); const header = values[0]; const otherMonths = values.slice(1).filter(r => !(Number(r[0]) === targetY && Number(r[1]) === targetM)); s.clearContents(); s.appendRow(header); otherMonths.forEach(r => s.appendRow(r)); data.shifts.forEach(r => s.appendRow([r.year, r.month, r.staffId, r.day, r.value])); } } if (action === 'saveRules') { let s = ss.getSheetByName('ルール') || ss.insertSheet('ルール'); s.clearContents(); s.appendRow(['dow','minF','minM','minTotal','required']); data.rules.forEach(r => s.appendRow([r.dow, r.minF, r.minM, r.minTotal || 0, r.required.join(',')])); } if (action === 'kibouKyuu') { // 種別（有給・欠勤など）と日付を文字列として明示的に保存する。 // 日付をそのまま書き込むとスプレッドシートが自動で日付型に // 変換してしまい、後で読み込むときにズレる原因になるため、 // セルの表示形式を「文字列（プレーンテキスト）」に固定する。 let s = ss.getSheetByName('希望休') || ss.insertSheet('希望休'); if (s.getLastRow() === 0) { s.appendRow(['送信日時','名前','対象月','種別','希望休日','備考']); } const row = s.getLastRow() + 1; const range = s.getRange(row, 1, 1, 6); range.setNumberFormats([['@','@','@','@','@','@']]); range.setValues([[ Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss'), data.name, data.month, data.type || 'off', data.dates.join(','), data.memo || '' ]]); } if (action === 'deleteKibou') { // 指定された行番号(1始まり、ヘッダー含む)の行を削除する const s = ss.getSheetByName('希望休'); if (s && data.rowIndex > 1) s.deleteRow(data.rowIndex); } if (action === 'updateKibou') { // 指定された行を書き換える const s = ss.getSheetByName('希望休'); if (s && data.rowIndex > 1) { const range = s.getRange(data.rowIndex, 1, 1, 6); range.setNumberFormats([['@','@','@','@','@','@']]); range.setValues([[ s.getRange(data.rowIndex, 1).getValue(), // 送信日時はそのまま data.name, data.month, data.type || 'off', data.dates.join(','), data.memo || '' ]]); } } return ContentService.createTextOutput(JSON.stringify({result:'success'})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = e.parameter.action; let result = {}; if (action === 'loadStaff') { const s = ss.getSheetByName('スタッフ'); if (s && s.getLastRow() > 1) { result.staff = s.getDataRange().getValues().slice(1).map(r => ({ id: Number(r[0]), name: String(r[1]).trim(), gender: r[2], type: r[3], days: r[4] ? r[4].toString().split(',').map(Number) : [] })); } else { result.staff = []; } } if (action === 'loadShifts') { const s = ss.getSheetByName('シフト'); if (s && s.getLastRow() > 1) { result.shifts = s.getDataRange().getValues().slice(1).map(r => ({ year: Number(r[0]), month: Number(r[1]), staffId: Number(r[2]), day: Number(r[3]), value: r[4] })); } else { result.shifts = []; } } if (action === 'loadRules') { const s = ss.getSheetByName('ルール'); if (s && s.getLastRow() > 1) { result.rules = s.getDataRange().getValues().slice(1).map(r => ({ dow: Number(r[0]), minF: Number(r[1]), minM: Number(r[2]), minTotal: Number(r[3]) || 0, required: r[4] ? r[4].toString().split(',').map(Number).filter(n => !isNaN(n) && n > 0) : [] })); } else { result.rules = []; } } if (action === 'loadKibou') { const s = ss.getSheetByName('希望休'); if (s && s.getLastRow() > 1) { const rows = s.getDataRange().getValues().slice(1).filter(r => r[1]); result.kibou = rows.map((r, i) => { const dCell = String(r[3]).trim(); const isNewFormat = LEAVE_TYPE_CODES.indexOf(dCell) !== -1; if (isNewFormat) { return { rowIndex: i + 2, // ヘッダー行(1行目)を除いた実際の行番号 timestamp: r[0], name: String(r[1]).trim(), month: String(r[2]).trim(), type: dCell, dates: parseDatesCell(r[4]), memo: r[5] }; } return { rowIndex: i + 2, timestamp: r[0], name: String(r[1]).trim(), month: String(r[2]).trim(), type: 'off', dates: parseDatesCell(r[3]), memo: r[4] }; }); } else { result.kibou = []; } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } // 希望休日のセルを安全にパースする。 // 過去に日付型として誤って保存されてしまった行が残っていても、 // 落ちずに正しい日付文字列（yyyy-MM-dd）に変換できるようにする。 function parseDatesCell(v) { if (!v) return []; if (v instanceof Date) { return [Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd')]; } return v.toString().split(',').map(x => x.trim()).filter(Boolean); }
